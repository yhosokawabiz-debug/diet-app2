🚀 初期セットアップ

Google スプレッドシートと連携してデータを安全に保存しましょう

1スプレッドシート作成

下のボタンをクリックして、専用のスプレッドシートを作成してください

2Google Apps Script 設定

作成したスプレッドシートで、以下の手順を実行してください：

  1. 「拡張機能」→「Apps Script」をクリック
  2. 既存のコードを削除
  3. 下のコードをコピーして貼り付け
  4. 「保存」をクリック
  5. 「デプロイ」→「新しいデプロイ」をクリック
  6. 「種類」を「ウェブアプリ」に設定
  7. 「次のユーザーとして実行」を「自分」に設定
  8. 「アクセスできるユーザー」を「全員」に設定
  9. 「デプロイ」をクリック
  10. 生成されたURLをコピー

const action = data.action; if (action === 'saveSettings') { settingsSheet.getRange('A1:B4').setValues([ ['スタート体重', data.startWeight], ['目標体重', data.targetWeight], ['開始日', data.startDate], ['目標期日', data.targetDate] ]); return ContentService.createTextOutput(JSON.stringify({success: true})); } if (action === 'getSettings') { const values = settingsSheet.getRange('A1:B4').getValues(); const settings = {}; values.forEach(row => { if (row[0] === 'スタート体重') settings.startWeight = row[1]; if (row[0] === '目標体重') settings.targetWeight = row[1]; if (row[0] === '開始日') settings.startDate = row[1]; if (row[0] === '目標期日') settings.targetDate = row[1]; }); return ContentService.createTextOutput(JSON.stringify(settings)); } if (action === 'addRecord') { const lastRow = recordsSheet.getLastRow(); const existingData = recordsSheet.getRange('A:B').getValues(); let updated = false; for (let i = 1; i < existingData.length; i++) { if (existingData[i][0] && new Date(existingData[i][0]).toDateString() === new Date(data.date).toDateString()) { recordsSheet.getRange(i + 1, 2).setValue(data.weight); updated = true; break; } } if (!updated) { recordsSheet.appendRow([new Date(data.date), data.weight]); } return ContentService.createTextOutput(JSON.stringify({success: true})); } if (action === 'getRecords') { const data = recordsSheet.getRange('A:B').getValues(); const records = []; for (let i = 1; i < data.length;
